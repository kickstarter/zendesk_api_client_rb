def options
  {
    :headers => {
      :accept => 'application/json',
      :accept_encoding => 'gzip;q=1.0,deflate;q=0.6,identity;q=0.3',
      :user_agent => "ZendeskAPI API #{ZendeskAPI::VERSION}"
    },
    :url => @url
  }.merge(client_options)
end
